N = 1
T = 100
I_min = 100000 # 

t1 = time.time()
I_ave_list = []
for i in range(nx.number_of_nodes(G)): # 
    vaccinated = i
    I_num_list = []
    for _ in range(T):
        node = random.randint(0,nx.number_of_nodes(G)-1)
        if node == vaccinated: # 
            I_num_list.append(0) # 0
        else:
            I_list = [node]
            I_list.extend(list(G.adj[node]))
            I_list = list(set(I_list))
            if vaccinated in list(G.adj[node]): # 
                I_list.remove(vaccinated) # 
            search_list = I_list[:]
            search_list.remove(node)
            for v in search_list:
                I_list.extend(list(G.adj[v]))
                I_list = list(set(I_list))
                if vaccinated in I_list: # 2
                    I_list.remove(vaccinated) # 
            I_num_list.append(len(I_list))
    I_ave = np.average(I_num_list)
    I_ave_list.append(I_ave)
    if I_min > I_ave:
        I_min = I_ave
        vac_min = vaccinated
t2 = time.time()
print(f"{t2-t1}")
print(f"{vac_min}  {I_min}")

data = []
data.append([N,vac_min,I_min,t2-t1])
